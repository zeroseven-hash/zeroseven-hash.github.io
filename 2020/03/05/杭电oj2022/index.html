杭电oj第2022题

比较简单，用二位数组储存数据，然后遍历记录下最大值位置


#include <cstdio>
#include<cmath>



int main(void)
{
    int n,m;
    while(scanf("%d %d",&m,&n)!=EOF)
    {
        int i,j;
        int a[100][100];
        int maxi=0,maxj=0;
        for(i=0;i<m;i++)
        {
            for(j=0;j<n;j++)
            {
                scanf("%d",&a[i][j]);
            }
        }
        for(i=0;i<m;i++)
        {
            for(j=0;j<n;j++)
            {
                if(abs(a[i][j])>abs(a[maxi][maxj]))
                {
                    maxi=i;
                    maxj=j;
                }
            }
        }
        printf("%d %d %d\n",maxi+1,maxj+1,a[maxi][maxj]);
    }
    return 0;
}
